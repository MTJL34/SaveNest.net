CREATE DATABASE IF NOT EXISTS savenest_db
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_unicode_ci;

USE savenest_db;

-- =====================
-- Roles
-- =====================
CREATE TABLE roles (
  id_roles INT UNSIGNED AUTO_INCREMENT,
  roles_name VARCHAR(50) NOT NULL,
  right_ BOOLEAN NOT NULL DEFAULT 0,
  PRIMARY KEY (id_roles),
  UNIQUE KEY uq_roles_name (roles_name)
) ENGINE=InnoDB;

-- =====================
-- Language
-- =====================
CREATE TABLE language_ (
  id_language INT UNSIGNED AUTO_INCREMENT,
  language_name VARCHAR(50) NOT NULL,
  PRIMARY KEY (id_language),
  UNIQUE KEY uq_language_name (language_name)
) ENGINE=InnoDB;

-- =====================
-- Country
-- =====================
CREATE TABLE country (
  id_country INT UNSIGNED AUTO_INCREMENT,
  country_name VARCHAR(50) NOT NULL,
  PRIMARY KEY (id_country),
  UNIQUE KEY uq_country_name (country_name)
) ENGINE=InnoDB;

-- =====================
-- SaveNest (profil/compte)
-- =====================
CREATE TABLE savenest (
  id_savenest INT UNSIGNED AUTO_INCREMENT,
  date_inscription DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id_savenest)
) ENGINE=InnoDB;

-- =====================
-- User
-- =====================
CREATE TABLE user_ (
  id_user INT UNSIGNED AUTO_INCREMENT,
  pseudo VARCHAR(50) NOT NULL,
  mail VARCHAR(150) NOT NULL,
  password VARCHAR(255) NOT NULL, -- hash (bcrypt/argon2)
  id_savenest INT UNSIGNED NOT NULL,
  id_country INT UNSIGNED NOT NULL,
  id_roles INT UNSIGNED NOT NULL,
  PRIMARY KEY (id_user),
  UNIQUE KEY uq_user_pseudo (pseudo),
  UNIQUE KEY uq_user_mail (mail),
  KEY idx_user_savenest (id_savenest),
  KEY idx_user_country (id_country),
  KEY idx_user_roles (id_roles),
  CONSTRAINT fk_user_savenest
    FOREIGN KEY (id_savenest) REFERENCES savenest(id_savenest)
    ON UPDATE CASCADE ON DELETE RESTRICT,
  CONSTRAINT fk_user_country
    FOREIGN KEY (id_country) REFERENCES country(id_country)
    ON UPDATE CASCADE ON DELETE RESTRICT,
  CONSTRAINT fk_user_roles
    FOREIGN KEY (id_roles) REFERENCES roles(id_roles)
    ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB;

-- =====================
-- Favs
-- =====================
CREATE TABLE favs (
  id_favs INT UNSIGNED AUTO_INCREMENT,
  title_favs VARCHAR(150) NOT NULL,
  url_favs VARCHAR(500) NOT NULL,
  added_date DATE NOT NULL DEFAULT (CURRENT_DATE),
  logo VARCHAR(150),
  PRIMARY KEY (id_favs),
  UNIQUE KEY uq_favs_title (title_favs),
  UNIQUE KEY uq_favs_url (url_favs)
) ENGINE=InnoDB;

-- =====================
-- Category
-- =====================
CREATE TABLE category (
  id_category INT UNSIGNED AUTO_INCREMENT,
  category_name VARCHAR(50) NOT NULL,
  confidentiality BOOLEAN NOT NULL DEFAULT 0,
  password VARCHAR(255) NULL, -- si catégorie confidentielle -> stocker hash aussi
  id_user INT UNSIGNED NOT NULL,
  PRIMARY KEY (id_category),
  UNIQUE KEY uq_category_user_name (id_user, category_name), -- unique par utilisateur (mieux que global)
  KEY idx_category_user (id_user),
  CONSTRAINT fk_category_user
    FOREIGN KEY (id_user) REFERENCES user_(id_user)
    ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;

-- =====================
-- Speaking (N:N user <-> language)
-- =====================
CREATE TABLE speaking (
  id_user INT UNSIGNED NOT NULL,
  id_language INT UNSIGNED NOT NULL,
  PRIMARY KEY (id_user, id_language),
  KEY idx_speaking_language (id_language),
  CONSTRAINT fk_speaking_user
    FOREIGN KEY (id_user) REFERENCES user_(id_user)
    ON UPDATE CASCADE ON DELETE CASCADE,
  CONSTRAINT fk_speaking_language
    FOREIGN KEY (id_language) REFERENCES language_(id_language)
    ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;

-- =====================
-- Save (N:N category <-> favs)
-- =====================
CREATE TABLE save_ (
  id_category INT UNSIGNED NOT NULL,
  id_favs INT UNSIGNED NOT NULL,
  PRIMARY KEY (id_category, id_favs),
  KEY idx_save_favs (id_favs),
  CONSTRAINT fk_save_category
    FOREIGN KEY (id_category) REFERENCES category(id_category)
    ON UPDATE CASCADE ON DELETE CASCADE,
  CONSTRAINT fk_save_favs
    FOREIGN KEY (id_favs) REFERENCES favs(id_favs)
    ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;
